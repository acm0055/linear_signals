x = -1:.01:5;
y = (x >= -1) + (-2).*(x >= 1) + (x >=3);
b = (x + 1).*(x >= 1) + (-1 * x).*(x >= 0) - (x >= 2);

figure
subplot(2,1,1)
plot(x, y, 'linewidth', 2)
xlabel('Time (t)')
ylabel('u(t)')
title('1.4A')
grid on
axis([-1 5 -2 3])

subplot(2,1,2)
plot(x, b, 'linewidth', 2)
xlabel('Time (t)')
ylabel('r(t)')
title('1.4B')
grid on
axis([-1 5 -2 3])
